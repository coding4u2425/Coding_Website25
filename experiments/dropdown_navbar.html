HTML:

<!DOCTYPE html>
<html>
<body>

  <div class="drop-navbar">
    <a href="#home"> Home </a>
    <div class="dropdown">
      <button class="dropbtn">
       Products
       <i class="fa-solid fa-caret-down"> </i>
      </button>
      <div class="dropdown-content">
       <a href="#"> PC </a>
       <a href="#"> Laptop </a>
       <a href="#"> Speaker </a>
      </div>
    </div>
    <a href="#about us"> About Us </a>
    <a href="#contact"> Contact </a>
  </div>

</body>
</html>
